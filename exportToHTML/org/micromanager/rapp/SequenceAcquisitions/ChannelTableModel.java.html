86      @Override 
87      public Object getValueAt(int rowIndex, int columnIndex) { 
88         if (channels_ != null && rowIndex < channels_.size()) { 
89            if (columnIndex == 0) { 
90               return channels_.get(rowIndex).useChannel; 
91            } else if (columnIndex == 1) { 
92               return channels_.get(rowIndex).config; 
93            } else if (columnIndex == 2) { 
94               return channels_.get(rowIndex).exposure; 
95            } else if (columnIndex == 3) { 
96               return channels_.get(rowIndex).laser_exposure; 
97            } else if (columnIndex == 4) { 
98               return channels_.get(rowIndex).KillCell; 
99            } else if (columnIndex == 5) { 
100              return channels_.get(rowIndex).color; 
101           } 
102        } 
103        return null; 
104     }

108        return getValueAt(0, c).getClass(); 
109     }

111     @Override 
112     public void setValueAt(Object value, int row, int col) { 
113            if (row < channels_.size() && value != null) { 
114   
115            ChannelSpec channel = channels_.get(row); 
116            if (col == 0) { 
117               channel.useChannel = ((Boolean) value); 
118            } else if (col == 1) { 
119               channel.config = value.toString(); 
120               channel.exposure = exposurePrefs_.getDouble("Exposure_" + acqEng_.getChannelGroup() + "_" + channel.config, 10.0); 
121            } else if (col == 2) { 
122               channel.exposure = ((Double) value); 
123               exposurePrefs_.putDouble("Exposure_" + acqEng_.getChannelGroup() + "_" + channel.config,channel.exposure); 
124               if (options_.syncExposureMainAndMDA_) { 
125                  studio_.setChannelExposureTime(acqEng_.getChannelGroup(), 
126                          channel.config, channel.exposure); 
127               } 
128            } else if (col == 3) { 
129               channel.laser_exposure = ((Double) value); 
130            } else if (col == 4) { 
131              channel.KillCell = ((Boolean) value); 
132            } else if (col == 5) { 
133               channel.color = (Color) value; 
134            } 
135            acqEng_.setChannel(row, channel); 
136         } 
137     }

154     @Override 
155     public void tableChanged(TableModelEvent e) { 
156        int row = e.getFirstRow(); 
157        if (row < 0) { 
158           return; 
159        } 
160        int col = e.getColumn(); 
161        if (col < 0) { 
162           return; 
163        } 
164        ChannelSpec channel = channels_.get(row); 
165        TableModel model = (TableModel) e.getSource(); 
166   
167        if (col == 5) { 
168           Color color = (Color) model.getValueAt(row, col); 
169           colorPrefs_.putInt("Color_" + acqEng_.getChannelGroup() + "_" + channel.config, color.getRGB()); 
170        } 
171     }

181     /** 
182      * Adds a new channel to the list in the MDA window 
183      */ 
184     public void addNewChannel() { 
185        ChannelSpec channel = new ChannelSpec(); 
186        channel.config = ""; 
187        if (acqEng_.getChannelConfigs().length > 0) { 
188   
189           for (String config : acqEng_.getChannelConfigs()) { 
190              boolean unique = true; 
191              for (ChannelSpec chan : channels_) { 
192                 if (config.contentEquals(chan.config)) { 
193                    unique = false; 
194                 } 
195              } 
196              if (unique) { 
197                 channel.config = config; 
198                 break; 
199              } 
200           } 
201           if (channel.config.length() == 0) { 
202              ReportingUtils.showMessage("No more channels are available\nin this channel group."); 
203           } else { 
204              channel.color = new Color(colorPrefs_.getInt( 
205                      "Color_" + acqEng_.getChannelGroup() + "_" + 
206                      channel.config, Color.white.getRGB())); 
207              channel.exposure = exposurePrefs_.getDouble( 
208                      "Exposure_" + acqEng_.getChannelGroup() + "_" + 
209                      channel.config, 10.0); 
210              channels_.add(channel); 
211           } 
212        } 
213     }

221     /** 
222      * Used to change the order of the channels in the MDA window 
223      * @param rowIdx 
224      * @return 
225      */ 
226     public int rowDown(int rowIdx) { 
227        if (rowIdx >= 0 && rowIdx < channels_.size() - 1) { 
228           ChannelSpec channel = channels_.get(rowIdx); 
229           channels_.remove(rowIdx); 
230           channels_.add(rowIdx + 1, channel); 
231           return rowIdx + 1; 
232        } 
233        return rowIdx; 
234     } 
235   
236     /** 
237      * Used to change the order of the channels in the MDA window 
238      */ 
239     public int rowUp(int rowIdx) { 
240        if (rowIdx >= 1 && rowIdx < channels_.size()) { 
241           ChannelSpec channel = channels_.get(rowIdx); 
242           channels_.remove(rowIdx); 
243           channels_.add(rowIdx - 1, channel); 
244           return rowIdx - 1; 
245        } 
246        return rowIdx; 
247     }

253     /** 
254      * Remove all channels from the list which are not compatible with 
255      * the current acquisition settings 
256      */ 
257     public void cleanUpConfigurationList() { 
258        String config; 
259        for (Iterator<ChannelSpec> it = channels_.iterator(); it.hasNext();) { 
260           config = it.next().config; 
261           if (!config.contentEquals("") && !acqEng_.isConfigAvailable(config)) { 
262              it.remove(); 
263           } 
264        } 
265        fireTableStructureChanged(); 
266     }

268     /** 
269      * reports if the same channel name is used twice 
270      */ 
271     public boolean duplicateChannels() { 
272        for (int i = 0; i < channels_.size() - 1; i++) { 
273           for (int j = i + 1; j < channels_.size(); j++) { 
274              if (channels_.get(i).config.equals(channels_.get(j).config)) { 
275                 return true; 
276              } 
277           } 
278        } 
279        return false; 
280     }

282     /** 
283      * Updates the exposure time in the given preset  
284      *  
285      * @param channelGroup - if it does not match current channelGroup,  
286      * no action will be taken 
287      *  
288      * @param channel - preset for which to change exposire time 
289      * @param exposure - desired exposure time 
290      */ 
291     public void setChannelExposureTime(String channelGroup, String channel,  
292             double exposure) { 
293        if (!channelGroup.equals(acqEng_.getChannelGroup())) 
294           return; 
295        for (ChannelSpec ch : channels_) { 
296           if (ch.config.equals(channel)) { 
297              ch.exposure = exposure; 
298              this.fireTableDataChanged(); 
299           } 
300   
301        } 
302     }
