85       protected boolean imagesSaving(){ 
86    
87           if (saveFiles_) { 
88               File root = new File(rootName_); 
89               if (!root.canWrite()) { 
90                   int result = JOptionPane.showConfirmDialog(null, "The specified root directory\n" + root.getAbsolutePath() + "\ndoes not exist. Create it?", "Directory not found.", 0); 
91                   if (result != 0) { 
92                       ReportingUtils.showMessage("Acquire Image won't save"); 
93                       return false; 
94                   } 
95                   root.mkdirs(); 
96                   if (!root.canWrite()) { 
97                       ReportingUtils.showError("Unable to save data to selected location: check that location exists.\nAcquisition canceled."); 
98                       return false; 
99                   } 
100              } else if (!enoughDiskSpace()) { 
101                  ReportingUtils.showError("Not enough space on disk to save the requested image set; acquisition canceled."); 
102                  return false; 
103              } 
104          } 
105          return true; 
106      }

108      protected String runSeqAcquisition(SequenceSettings acquisitionSettings) { 
109          app_.enableLiveMode(false); 
110          ArrayList<ChannelSpec> channels =  acquisitionSettings.channels; 
111          String chanelGroup_ = acquisitionSettings.channelGroup; 
112   
113          try { 
114              core_.waitForDevice(core_.getCameraDevice()); 
115              Thread.sleep(100); // wait and start acquisition 
116          } catch (Exception e) {

118          } 
119          //for (ChannelSpec presetConfig : channels){

124          if (!isRunning_.get()) { 
125              stopAcqRequested_.set(false); 
126              SeqAcqGui.progressBar.setIndeterminate(true); 
127              Thread th = new Thread("Sequence Acquisition thread") { 
128                  @Override 
129                  public void run() { 
130                      try { 
131                          SeqAcqGui.acquireButton_.setEnabled(false); 
132                          isRunning_.set(true); 
133                          ImagePlus iPlus ;

135                          int progress = 0;

139                          for (ChannelSpec presetConfig : channels){ 
140                              if (stopAcqRequested_.get()) { 
141                                  ReportingUtils.showMessage("Acquisition Stop."); 
142                                  break; 
143                              } 
144   
145                              // Set the Chanel Exposure Time 
146                              app_.enableLiveMode(false); // Make sure the Live Mode is off 
147                              app_.setChannelExposureTime(chanelGroup_, presetConfig.config, presetConfig.exposure); 
148                              // Then Change The Chanel Config (Preset ) 
149                              core_.setConfig(chanelGroup_, presetConfig.config.toString()); 
150   
151                              // Take a Photo for each fix chanel 
152                              core_.snapImage(); 
153                              core_.getLastTaggedImage(); 
154                              iPlus = IJ.getImage(); 
155

157                                  System.out.println( "You Are :" + presetConfig.config); 
158                              } 
159   
160                              // if the Images was not save , we do the segmentation for the image in Memory 
161                              if(!saveFiles_  && presetConfig.doSegmentation){ 
162                                  ArrayList[] ll =  rappController_ref.brightFieldSegmenter(iPlus, presetConfig.config.toString()); 
163                                  if (presetConfig.KillCell) { 
164                                      app_.enableLiveMode(true); //  Open the live mode before shooting 
165                                      rappController_ref.shootFromSegmentationListPoint(ll); 
166                                  } 
167                              } 
168   
169                              if (saveFiles_ && !SeqAcqGui.saveMultiTiff_) { 
170                                  // The acquires Images are saving as separate Image. 
171                                  IJ.save(iPlus, rootName_ +  "\\"+ dirName_+ "_"+ presetConfig.config.toLowerCase() + ".tif"); 
172                                  if(presetConfig.doSegmentation){ 
173                                      ImagePlus image_ =   IJ.openImage(rootName_ +  "\\"+ dirName_+ "_"+ presetConfig.config.toLowerCase() + ".tif"); 
174                                      // Execute the Segmentation depends on the Colors. 
175                                      if (presetConfig.KillCell) { 
176                                          ArrayList[] ll =  rappController_ref.brightFieldSegmenter(image_, presetConfig.config.toString()); 
177                                          app_.enableLiveMode(true); //  Open the live mode before shooting 
178                                          rappController_ref.createMultiPointAndShootFromRoeList(); 
179                                          //rappController_ref.shootFromSegmentationListPoint(ll); 
180                                      } 
181                                  } 
182   
183                              } else if(saveFiles_ && SeqAcqGui.saveMultiTiff_){ 
184                                  // The acquires Images are saving as a stack Image. 
185                                  IJ.save(iPlus, rootName_ +  "\\"+ dirName_+ "_"+ presetConfig.config.toLowerCase() + ".tif"); 
186                                  IJ.open(rootName_ +  "\\"+ dirName_+ "_"+ presetConfig.config.toLowerCase() + ".tif"); 
187   
188                                  ImagePlus image_ =   IJ.openImage(rootName_ +  "\\"+ dirName_+ "_"+ presetConfig.config.toLowerCase() + ".tif"); 
189                                  if (presetConfig.KillCell) { 
190                                      ArrayList[] ll =  rappController_ref.brightFieldSegmenter(image_, presetConfig.config.toString()); 
191                                      app_.enableLiveMode(true); //  Open the live mode before shooting 
192                                      rappController_ref.shootFromSegmentationListPoint(ll); 
193                                  } 
194                              } 
195   
196   
197                              progress  +=  100 / channels.size(); 
198                              SeqAcqGui.progressBar.setValue(progress); 
199                              SeqAcqGui.taskOutput.append(String.format("Completed %d%% of Sequence Task.\n", progress)); 
200                          } 
201   
202                          if(saveFiles_ && SeqAcqGui.saveMultiTiff_){ 
203                              IJ.run("Images to Stack", "name=Stack title=[] use"); 
204                              IJ.saveAs("Tiff", rootName_ +  "\\"+ dirName_+ "_"+ "Stack" + ".tif"); 
205                          } 
206                          JOptionPane.showMessageDialog(IJ.getImage().getWindow(), "Sequence Acquisition " 
207                                  + (!stopAcqRequested_.get() ? "finished." : "canceled.")); 
208                      } catch (Exception e) { 
209                          ReportingUtils.showError(e); 
210                          e.printStackTrace(); 
211                          ReportingUtils.showMessage("Acquisition Stop Due to some Error," + 
212                                  " Please check your input data. "+ 
213                                  " We are Sorry about that" 
214                                  ); 
215                      } finally { 
216                          SwingUtilities.invokeLater((Runnable) () -> SeqAcqGui.progressBar.setIndeterminate(false)); 
217                          RappGui.getInstance().setCursor(null); // turn off the wait cursor 
218                          SeqAcqGui.acquireButton_.setEnabled(true); // Activate the Button again 
219                          isRunning_.set(false); 
220                          stopAcqRequested_.set(false); 
221                      } 
222                  } 
223              }; 
224              th.start(); 
225          } 
226          return null; 
227      }

307      protected boolean enoughDiskSpace() { 
308          File root = new File(this.rootName_); 
309   
310          do { 
311              if (root.exists()) { 
312                  long usableMB = root.getUsableSpace() / 1048576L; 
313                  return 1.25D * (double)this.getTotalMB() < (double)usableMB; 
314              } 
315   
316              root = root.getParentFile(); 
317          } while(root != null); 
318   
319          return false; 
320      }
