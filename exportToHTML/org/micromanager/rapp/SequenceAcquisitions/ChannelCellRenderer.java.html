13   /** 
14    * Renderer class for the channel table. 
15    */ 
16   public class ChannelCellRenderer extends JLabel implements TableCellRenderer { 
17    
18      private static final long serialVersionUID = -4328340719459382679L; 
19      private AcquisitionEngine acqEng_; 
20    
21      // This method is called each time a cell in a column 
22      // using this renderer needs to be rendered. 
23      public ChannelCellRenderer(AcquisitionEngine acqEng) { 
24         super(); 
25         acqEng_ = acqEng; 
26      } 
27    
28      @Override 
29      public Component getTableCellRendererComponent(JTable table, Object value, 
30              boolean isSelected, boolean hasFocus, int rowIndex, int colIndex) { 
31    
32         // https://stackoverflow.com/a/3055930 
33         if (value == null) { 
34            return null; 
35         } 
36         ChannelTableModel model = (ChannelTableModel) table.getModel(); 
37         ArrayList<ChannelSpec> channels = model.getChannels(); 
38         ChannelSpec channel = channels.get(rowIndex); 
39         this.setEnabled(table.isEnabled()); 
40    
41         colIndex = table.convertColumnIndexToModel(colIndex); 
42    
43         setOpaque(false); 
44         if (colIndex == 0) { 
45            JCheckBox check = new JCheckBox("", channel.useChannel); 
46            check.setEnabled(table.isEnabled()); 
47            check.setOpaque(true); 
48            if (isSelected) { 
49               check.setBackground(table.getSelectionBackground()); 
50               check.setOpaque(true); 
51            } else { 
52               check.setOpaque(false); 
53               check.setBackground(table.getBackground()); 
54            } 
55            return check; 
56         } else { 
57            if (colIndex == 1) { 
58               setText(channel.config); 
59            } else if (colIndex == 2) { 
60               setText(NumberUtils.doubleToDisplayString(channel.exposure)); 
61            } else if (colIndex == 3) { 
62               setText(NumberUtils.doubleToDisplayString(channel.laser_exposure)); 
63            } else { 
64               if (colIndex == 4) { 
65                  JCheckBox check = new JCheckBox("", channel.KillCell); 
66                  check.setEnabled(acqEng_.isDoSegmentationEnabled() && table.isEnabled()); 
67                  if (isSelected) { 
68                     check.setBackground(table.getSelectionBackground()); 
69                     check.setOpaque(true); 
70                  } else { 
71                     check.setOpaque(false); 
72                     check.setBackground(table.getBackground()); 
73                  } 
74                  return check; 
75               } else if (colIndex == 5) { 
76                  setText(""); 
77                  setBackground(channel.color); 
78                  setOpaque(true); 
79               } 
80            } 
81    
82    
83            if (isSelected) { 
84               setBackground(table.getSelectionBackground()); 
85               setOpaque(true); 
86            } else { 
87               setOpaque(false); 
88               setBackground(table.getBackground()); 
89            } 
90    
91            // Since the renderer is a component, return itself 
92            return this; 
93         } 
94      }
