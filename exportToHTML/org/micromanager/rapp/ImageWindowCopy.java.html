763      public void mouseWheelMoved(MouseWheelEvent event) { 
764          int rotation = event.getWheelRotation(); 
765          int width = this.imp.getWidth(); 
766          int height = this.imp.getHeight(); 
767          Rectangle srcRect = this.ic.getSrcRect(); 
768          int xstart = srcRect.x; 
769          int ystart = srcRect.y; 
770          if (!IJ.spaceBarDown() && srcRect.height != height) { 
771              srcRect.y += rotation * Math.max(height / 200, 1); 
772              if (srcRect.y < 0) { 
773                  srcRect.y = 0; 
774              } 
775   
776              if (srcRect.y + srcRect.height > height) { 
777                  srcRect.y = height - srcRect.height; 
778              } 
779          } else { 
780              srcRect.x += rotation * Math.max(width / 200, 1); 
781              if (srcRect.x < 0) { 
782                  srcRect.x = 0; 
783              } 
784   
785              if (srcRect.x + srcRect.width > width) { 
786                  srcRect.x = width - srcRect.width; 
787              } 
788          } 
789   
790          if (srcRect.x != xstart || srcRect.y != ystart) { 
791              this.ic.repaint(); 
792          } 
793   
794      }
