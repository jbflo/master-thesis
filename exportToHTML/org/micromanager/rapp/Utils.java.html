15      public static List<Polygon> FloatToNormalPolygon(List<FloatPolygon> floatPolygons) { 
16         // manually copy from FloatPolygon to Polygon 
17         List<Polygon> roiPolygons = new ArrayList<Polygon>(); 
18         for (FloatPolygon fp : floatPolygons) { 
19            Polygon p = new Polygon(); 
20            for (int i = 0; i < fp.npoints; i++) { 
21               p.addPoint((int) (0.5 + fp.xpoints[i]), 
22                       (int) (0.5 + fp.ypoints[i])); 
23            } 
24            roiPolygons.add(p); 
25         } 
26         return roiPolygons; 
27      }
