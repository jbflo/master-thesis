21       public String ChooseFileDialog(String title) { 
22    
23           JFileChooser fileChooser = new JFileChooser(); 
24           fileChooser.setDialogTitle(title); 
25           File choose_file; 
26           int userSelection = fileChooser.showSaveDialog(RappGui.getInstance()); 
27           if (userSelection == JFileChooser.APPROVE_OPTION) { 
28               choose_file = fileChooser.getSelectedFile(); 
29               System.out.println("Save as file: " + choose_file.getAbsolutePath()); 
30               return choose_file.getAbsolutePath(); 
31           } 
32           return null ;

34       }
