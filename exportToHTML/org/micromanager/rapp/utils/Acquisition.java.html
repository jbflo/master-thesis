179      private void saveAcqToOMETiff(IFormatWriter writer, String acq, int length) 
180              throws Exception { 
181          ImageCache imgCache = gui_.getAcquisitionImageCache(acq); 
182          for (int ind = 0; ind < length; ind++) { 
183              TaggedImage timg = imgCache.getImage(ind, 0, 0, 0); // returns null if no image exists at these coordinates 
184              Object img = timg.pix; 
185              if (img instanceof byte[]) { 
186                  System.out.println("Img is in bytes"); 
187                  writer.saveBytes(ind, (byte[]) img); 
188              } else if (img instanceof short[]) { 
189  //                byte[] bytes = DataTools.shortsToBytes((short[]) img, true); 
190                  short[] img2 = (short[]) img; 
191                  byte[] bytes = DataTools.shortsToBytes(img2, false); 
192                  System.out.println("Img is short[], converting to bytes, i = " + ind); 
193  //                writer.savePlane(ind,img); 
194                  writer.saveBytes(ind, bytes); 
195              } else { 
196                  System.out.println("I don't know what type img is!"); 
197              } 
198          } 
199   
200      }

203      void saveLayersToOMETiff(IFormatWriter writer, Object img, int layer) 
204              throws Exception { 
205  //        Object img = core_.getImage(); 
206          if (img instanceof byte[]) { 
207              System.out.println("Img is in bytes"); 
208              writer.saveBytes(layer, (byte[]) img); 
209          } else if (img instanceof short[]) { 
210              byte[] bytes = DataTools.shortsToBytes((short[]) img, true); 
211  //            System.out.println("Img is short[], converting to bytes, i = " + layer); 
212              writer.saveBytes(layer, bytes); 
213          } else  if (img instanceof int[]){ 
214              byte[] bytes = DataTools.intsToBytes((int[]) img, true); 
215              writer.saveBytes(layer, bytes); 
216          } else 
217          { 
218              System.out.println("I don't know what type img is!"); 
219          } 
220      }

222      public IFormatWriter generateWriter(String path, OMEXMLMetadata m) 
223              throws FormatException, IOException { 
224          IFormatWriter writer = new ImageWriter().getWriter(path); 
225          writer.setWriteSequentially(true); 
226          writer.setMetadataRetrieve(m); 
227          writer.setCompression("LZW"); 
228          writer.setId(path); 
229   
230          return writer; 
231      }

233      private PositiveFloat checkPixelPitch() { 
234   
235          PositiveFloat pitch = new PositiveFloat(1.0); 
236   
237          try { 
238              String binningStr = core_.getProperty(core_.getCameraDevice(), "Binning"); 
239              float binning = 1; 
240              if (binningStr.equals("1")) { 
241                  binning = 1; 
242              } else if (binningStr.equals("2")) { 
243                  binning = 2; 
244              } else if (binningStr.equals("4")) { 
245                  binning = 4; 
246              } else if (binningStr.equals("8")) { 
247                  binning = 8; 
248              } 
249   
250              pitch = new PositiveFloat(binning * 6.45); 
251   
252          } catch (Exception e) { 
253              System.out.println(e.getMessage()); 
254          } 
255   
256          return pitch; 
257      }

260      public OMEXMLMetadata setBasicMetadata(ArrayList<String> delays) 
261              throws ServiceException { 
262   
263          OMEXMLServiceImpl serv = new OMEXMLServiceImpl(); 
264          OMEXMLMetadata m = serv.createOMEXMLMetadata(); 
265   
266          int no_delays = delays.size(); 
267          // delays must be parsed as strings for metadata 
268          String[] delArrayStr = new String[no_delays]; 
269          for (int ind = 0; ind < no_delays; ind++) { 
270              delArrayStr[ind] = String.valueOf(delays.get(ind)); 
271          } 
272   
273          try { 
274              m.createRoot(); 
275              m.setImageID("Image:0", 0); 
276              m.setPixelsID("Pixels:0", 0); 
277              m.setPixelsDimensionOrder(DimensionOrder.XYZCT, 0); 
278              m.setChannelID("Channel:0:0", 0, 0); 
279              m.setChannelSamplesPerPixel(new PositiveInteger(1), 0, 0); 
280              m.setPixelsBinDataBigEndian(Boolean.FALSE, 0, 0); 
281  //            m.setPixelsType(PixelType.UINT8, 0); 
282  //            m.setImageDescription(sas.toString(), 0); 
283   
284              long bpp = core_.getBytesPerPixel(); 
285   
286   
287              if (bpp == 1) { 
288                  m.setPixelsType(PixelType.UINT8, 0); 
289              } 
290              if (bpp == 2) { 
291                  m.setPixelsType(PixelType.UINT16, 0); 
292              } 
293   
294              PositiveInteger w1 = new PositiveInteger((int) core_.getImageWidth()); 
295              PositiveInteger h1 = new PositiveInteger((int) core_.getImageHeight()); 
296              PositiveInteger g1 = new PositiveInteger(no_delays); 
297   
298              m.setPixelsSizeX((w1), 0); 
299              m.setPixelsSizeY((h1), 0); 
300              m.setPixelsSizeZ(new PositiveInteger(1), 0); 
301              m.setPixelsSizeC(new PositiveInteger(1), 0); 
302              m.setPixelsSizeT(g1, 0); 
303   
304              PositiveFloat pitch = checkPixelPitch(); 
305              double pitchD = pitch.getValue(); 
306              Length len = new Length(1,ome.units.UNITS.MICROM); 
307              m.setPixelsPhysicalSizeX(new Length(pitchD,ome.units.UNITS.MICROM),0); 
308              m.setPixelsPhysicalSizeY(new Length(pitchD,ome.units.UNITS.MICROM),0); 
309              m.setPixelsPhysicalSizeZ(new Length(1.0,ome.units.UNITS.MICROM), 0); 
310   
311              PlatformIndependentGuidGen p = PlatformIndependentGuidGen.getInstance(); 
312   
313              for (int ii = 0; ii < no_delays; ii++) { 
314   
315                  m.setUUIDFileName(delArrayStr[ii], 0, ii); 
316                  m.setUUIDValue(p.genNewGuid(), 0, ii); 
317                  m.setTiffDataPlaneCount(new NonNegativeInteger(0), 0, ii); 
318                  m.setTiffDataIFD(new NonNegativeInteger(0), 0, ii); 
319                  m.setTiffDataFirstZ(new NonNegativeInteger(0), 0, ii); 
320                  m.setTiffDataFirstC(new NonNegativeInteger(0), 0, ii); 
321                  m.setTiffDataFirstT(new NonNegativeInteger(0), 0, ii); 
322                  m.setPlaneTheC(new NonNegativeInteger(0), 0, ii); 
323                  m.setPlaneTheZ(new NonNegativeInteger(0), 0, ii); 
324                  m.setPlaneTheT(new NonNegativeInteger(ii), 0, ii); 
325                  m.setTiffDataPlaneCount(new NonNegativeInteger(ii), 0, ii); 
326                  System.out.println("done loop ind " + ii); 
327              } 
328   
329              // deal FLIMfit issue loading single plane images with moduloAlongT 
330              if (no_delays > 2){ 
331                  CoreMetadata cm = new CoreMetadata(); 
332   
333                  cm.moduloT.labels = delArrayStr; 
334                  cm.moduloT.unit = "ps"; 
335                  cm.moduloT.typeDescription = "Gated"; 
336                  cm.moduloT.type = loci.formats.FormatTools.LIFETIME; 
337                  serv.addModuloAlong(m, cm, 0); 
338                  System.out.println("did addModulo"); 
339              } 
340   
341          } catch (Exception e) { 
342              System.out.println(e.getMessage()); 
343          } 
344   
345          return m; 
346      }
