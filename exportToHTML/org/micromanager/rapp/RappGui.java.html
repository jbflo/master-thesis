744      private JToggleButton createJButton(String text) { 
745          JToggleButton button = new JToggleButton(text); 
746          button.setForeground(Color.decode("#ecf0f1")); 
747          button.setBorder(null); 
748          button.setFont(new java.awt.Font("Tahoma", 0, 15)); // NOI18N 
749          button.setBorderPainted(false); 
750          button.setFocusPainted(false); 
751          button.setContentAreaFilled(false); 
752          ImageIcon icon = new ImageIcon(path.concat("Resources/Images/" + text + ".png")); 
753          Image img = icon.getImage(); 
754          Image new_img = img.getScaledInstance(20, 20, java.awt.Image.SCALE_SMOOTH); 
755          button.setIcon(new ImageIcon(new_img)); 
756          button.setSelectedIcon(new ImageIcon(new_img)); 
757          return button; 
758      }

793      private void confirmQuit() { 
794          int n = JOptionPane.showConfirmDialog(appInterface_, 
795                  "Quit: are you sure to quit?", "Quit", JOptionPane.YES_NO_OPTION); 
796          if (n == JOptionPane.YES_OPTION) { 
797              LiveMode_btn.setSelected(false); 
798              dispose(); 
799              studio_.makeActive(); 
800          } 
801          else  GUIUtils.recallPosition(appInterface_); 
802      }
