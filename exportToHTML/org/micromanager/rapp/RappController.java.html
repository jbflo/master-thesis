148      /** 
149       * Gets the vectorial mean of an array of Points. 
150       */ 
151       private static Point2D.Double meanPosition2D(Point[] points) { 
152           double xsum = 0; 
153           double ysum = 0; 
154           int n = points.length; 
155           for (int i = 0; i < n; ++i) { 
156              xsum += points[i].x; 
157              ysum += points[i].y; 
158           } 
159           return new Point2D.Double(xsum/n, ysum/n); 
160       }

239      /** 
240       * Sets the Channel Group to the targeting channel, if it exists. 
241       * @return 
242       */ 
243      public Configuration prepareChannel() { 
244          Configuration originalConfig = null; 
245          String channelGroup = core_.getChannelGroup(); 
246          try { 
247              if (targetingChannel_ != null && targetingChannel_.length() > 0) { 
248                  originalConfig = core_.getConfigGroupState(channelGroup); 
249                  if (!originalConfig.isConfigurationIncluded(core_.getConfigData(channelGroup, targetingChannel_))) { 
250                      if (app_.isAcquisitionRunning()) { 
251                          app_.setPause(true); 
252                      } 
253                      core_.setConfig(channelGroup, targetingChannel_); 
254                  } 
255              } 
256          } catch (Exception ex) { 
257              ReportingUtils.logError(ex); 
258          } 
259          return originalConfig; 
260      }

263       * Should be called with the value returned by prepareChannel. 
264       * Returns Channel Group to its original settings, if needed. 
265       * @param originalConfig value returned by prepareChannel

268          if (originalConfig != null) { 
269              try { 
270                  core_.setSystemState(originalConfig); 
271                  if (app_.isAcquisitionRunning() && app_.isPaused()) { 
272                      app_.setPause(false);

274              } catch (Exception ex) { 
275                  ReportingUtils.logError(ex); 
276              } 
277          }

280      /** 
281       * Opens the targeting shutter, if it has been specified. 
282       * @return true if it was already open 
283       */ 
284      public boolean prepareShutter() { 
285          try { 
286              if (targetingShutter_ != null && targetingShutter_.length() > 0) { 
287                  boolean originallyOpen = core_.getShutterOpen(targetingShutter_); 
288                  if (!originallyOpen) { 
289                      core_.setShutterOpen(targetingShutter_, true); 
290                      core_.waitForDevice(targetingShutter_); 
291                  } 
292                  return originallyOpen; 
293              } 
294          } catch (Exception ex) { 
295              ReportingUtils.logError(ex); 
296          } 
297          return true; // by default, say it was already open 
298      }

416      // Find the brightest spot in an ImageProcessor. The image is first blurred 
417      // and then the pixel with maximum intensity is returned. 
418      private static Point findPeak(ImageProcessor proc) { 
419          ImageProcessor blurImage = proc.duplicate(); 
420          blurImage.setRoi((Roi) null); 
421          GaussianBlur blur = new GaussianBlur(); 
422          blur.blurGaussian(blurImage, 10, 10, 0.01); 
423          //showProcessor("findPeak",proc); 
424          Point x = ImageUtils.findMaxPixel(blurImage); 
425          x.translate(1, 1); 
426          return x; 
427      }

495      /** 
496       * Illuminates and images five control points near the center, 
497       * and return an affine transform mapping from image coordinates 
498       * to phototargeter coordinates. 
499       */ 
500      private AffineTransform generateLinearMapping() { 
501          double centerX = dev_.getXRange() / 2 + dev_.getXMinimum(); 
502          double centerY = dev_.getYRange() / 2 + dev_.getYMinimum(); 
503          double spacing = Math.min(dev_.getXRange(), dev_.getYRange()) / 25;  // use 4% of galvo/SLM range 
504          Map<Point2D.Double, Point2D.Double> spotMap 
505                  = new HashMap<Point2D.Double, Point2D.Double>(); 
506   
507          measureAndAddToSpotMap(spotMap, new Point2D.Double(centerX, centerY)); 
508          measureAndAddToSpotMap(spotMap, new Point2D.Double(centerX, centerY + spacing)); 
509          measureAndAddToSpotMap(spotMap, new Point2D.Double(centerX + spacing, centerY)); 
510          measureAndAddToSpotMap(spotMap, new Point2D.Double(centerX, centerY - spacing)); 
511          measureAndAddToSpotMap(spotMap, new Point2D.Double(centerX - spacing, centerY)); 
512          if (stopRequested_.get()) { 
513              return null; 
514          } 
515          try { 
516              // require that the RMS value between the mapped points and the measured points be less than 5% of image size 
517              // also require that the RMS value be less than the spacing between points in the galvo/SLM coordinate system 
518              // (2nd requirement was probably the intent of the code until r15505, but parameters were interchanged in call) 
519              final long imageSize = Math.min(core_.getImageWidth(), core_.getImageHeight()); 
520              return MathFunctions.generateAffineTransformFromPointPairs(spotMap, imageSize*0.05, spacing); 
521          } catch (Exception e) { 
522              throw new RuntimeException("Spots aren't detected as expected. Is DMD in focus and roughly centered in camera's field of view?"); 
523          } 
524      }

526      /** 
527       * Generate a nonlinear calibration mapping for the current device settings. 
528       * A rectangular lattice of points is illuminated one-by-one on the 
529       * projection device, and locations in camera pixels of corresponding 
530       * spots on the camera image are recorded. For each rectangular 
531       * cell in the grid, we take the four point mappings (camera to projector) 
532       * and generate a local AffineTransform using linear least squares. 
533       * Cells with suspect measured corner positions are discarded. 
534       * A mapping of cell polygon to AffineTransform is generated. 
535       */ 
536      private Map<Polygon, AffineTransform> generateNonlinearMapping() { 
537   
538          // get the affine transform near the center spot 
539          final AffineTransform firstApproxAffine = generateLinearMapping(); 
540   
541          // then use this single transform to estimate what SLM coordinates 
542          // correspond to the image's corner positions 
543          final Point2D.Double camCorner1 = (Point2D.Double) firstApproxAffine.transform(new Point2D.Double(0, 0), null); 
544          final Point2D.Double camCorner2 = (Point2D.Double) firstApproxAffine.transform(new Point2D.Double((int) core_.getImageWidth(), (int) core_.getImageHeight()), null); 
545          final Point2D.Double camCorner3 = (Point2D.Double) firstApproxAffine.transform(new Point2D.Double(0, (int) core_.getImageHeight()), null); 
546          final Point2D.Double camCorner4 = (Point2D.Double) firstApproxAffine.transform(new Point2D.Double((int) core_.getImageWidth(), 0), null); 
547   
548          // figure out camera's bounds in SLM coordinates 
549          // min/max because we don't know the relative orientation of the camera and SLM 
550          // do some extra checking in case camera/SLM aren't at exactly 90 degrees from each other, 
551          // but still better that they are at 0, 90, 180, or 270 degrees from each other 
552          // TODO can create grid along camera location instead of SLM's if camera is the limiting factor; this will make arbitrary rotation possible 
553          final double camLeft = Math.min(Math.min(Math.min(camCorner1.x, camCorner2.x), camCorner3.x), camCorner4.x); 
554          final double camRight = Math.max(Math.max(Math.max(camCorner1.x, camCorner2.x), camCorner3.x), camCorner4.x); 
555          final double camTop = Math.min(Math.min(Math.min(camCorner1.y, camCorner2.y), camCorner3.y), camCorner4.y); 
556          final double camBottom = Math.max(Math.max(Math.max(camCorner1.y, camCorner2.y), camCorner3.y), camCorner4.y); 
557   
558          // these are the SLM's bounds 
559          final double slmLeft = dev_.getXMinimum(); 
560          final double slmRight = dev_.getXRange() + dev_.getXMinimum(); 
561          final double slmTop = dev_.getYMinimum(); 
562          final double slmBottom = dev_.getYRange() + dev_.getYMinimum(); 
563   
564          // figure out the "overlap region" where both the camera and SLM 
565          // can "see", expressed in SLM coordinates 
566          final double left = Math.max(camLeft, slmLeft); 
567          final double right = Math.min(camRight, slmRight); 
568          final double top = Math.max(camTop, slmTop); 
569          final double bottom = Math.min(camBottom, slmBottom); 
570          final double width = right - left; 
571          final double height = bottom - top; 
572   
573          // compute a grid of SLM points inside the "overlap region" 
574          // nGrid is how many polygons in both X and Y 
575          // require (nGrid + 1)^2 spot measurements to get nGrid^2 squares 
576          // TODO allow user to change nGrid 
577          final int nGrid = 7; 
578          Point2D.Double slmPoint[][] = new Point2D.Double[1 + nGrid][1 + nGrid]; 
579          Point2D.Double camPoint[][] = new Point2D.Double[1 + nGrid][1 + nGrid]; 
580   
581          // tabulate the camera spot at each of SLM grid points 
582          for (int i = 0; i <= nGrid; ++i) { 
583              for (int j = 0; j <= nGrid; ++j) { 
584                  double xoffset = ((i + 0.5) * width / (nGrid + 1.0)); 
585                  double yoffset = ((j + 0.5) * height / (nGrid + 1.0)); 
586                  slmPoint[i][j] = new Point2D.Double(left + xoffset, top + yoffset); 
587                  Point spot = measureSpotOnCamera(slmPoint[i][j], true); 
588                  if (spot != null) { 
589                      camPoint[i][j] = toDoublePoint(spot); 
590                  } 
591              } 
592          } 
593   
594          if (stopRequested_.get()) { 
595              return null; 
596          } 
597   
598          // now make a grid of (square) polygons (in camera's coordinate system) 
599          // and generate an affine transform for each of these square regions 
600          Map<Polygon, AffineTransform> bigMap 
601                  = new HashMap<Polygon, AffineTransform>(); 
602          for (int i = 0; i <= nGrid - 1; ++i) { 
603              for (int j = 0; j <= nGrid - 1; ++j) { 
604                  Polygon poly = new Polygon(); 
605                  addVertex(poly, toIntPoint(camPoint[i][j])); 
606                  addVertex(poly, toIntPoint(camPoint[i][j + 1])); 
607                  addVertex(poly, toIntPoint(camPoint[i + 1][j + 1])); 
608                  addVertex(poly, toIntPoint(camPoint[i + 1][j])); 
609   
610                  Map<Point2D.Double, Point2D.Double> map 
611                          = new HashMap<Point2D.Double, Point2D.Double>(); 
612                  map.put(camPoint[i][j], slmPoint[i][j]); 
613                  map.put(camPoint[i][j + 1], slmPoint[i][j + 1]); 
614                  map.put(camPoint[i + 1][j], slmPoint[i + 1][j]); 
615                  map.put(camPoint[i + 1][j + 1], slmPoint[i + 1][j + 1]); 
616                  double srcDX = Math.abs((camPoint[i+1][j].x - camPoint[i][j].x))/4; 
617                  double srcDY = Math.abs((camPoint[i][j+1].y - camPoint[i][j].y))/4; 
618                  double srcTol = Math.max(srcDX, srcDY); 
619   
620                  try { 
621                      AffineTransform transform = MathFunctions.generateAffineTransformFromPointPairs(map, srcTol, Double.MAX_VALUE); 
622                      bigMap.put(poly, transform); 
623                  } catch (Exception e) { 
624                      ReportingUtils.logError("Bad cell in mapping."); 
625                  } 
626              } 
627          } 
628          return bigMap; 
629      }

699      // ## Transforming points according to a nonlinear calibration mapping. 
700   
701      // Transform a point, pt, given the mapping, which is a Map of polygon cells 
702      // to AffineTransforms. 
703      private static Point2D.Double transformPoint(Map<Polygon, AffineTransform> mapping, Point2D.Double pt) { 
704          Set<Polygon> set = mapping.keySet(); 
705          // First find out if the given point is inside a cell, and if so, 
706          // transform it with that cell's AffineTransform. 
707          for (Polygon poly : set) { 
708              if (poly.contains(pt)) { 
709                  return (Point2D.Double) mapping.get(poly).transform(pt, null); 
710              } 
711          } 
712          // The point isn't inside any cell, so search for the closest cell 
713          // and use the AffineTransform from that. 
714          double minDistance = Double.MAX_VALUE; 
715          Polygon bestPoly = null; 
716          for (Polygon poly : set) { 
717              double distance = meanPosition2D(getVertices(poly)).distance(pt.x, pt.y); 
718              if (minDistance > distance) { 
719                  bestPoly = poly; 
720                  minDistance = distance; 
721              } 
722          } 
723          if (bestPoly == null) { 
724              throw new RuntimeException("Unable to map point to device."); 
725          } 
726          return (Point2D.Double) mapping.get(bestPoly).transform(pt, null); 
727      }
